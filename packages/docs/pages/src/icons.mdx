import React from "react";
import { Meta, Title, IconGallery, IconItem } from "@storybook/addon-docs/blocks";
import * as icons from "@react-ck/icon/icons/All";
import { Icon, Input, Manager } from "react-ck";

<Meta title="Generic/Icon/Gallery" />

export function Demo ()  {
  
  const [search, setSearch] = React.useState("");
  const filteredIcons = Object.keys(icons).filter(function (i) {
    return i.toLowerCase().includes(search.toLowerCase())
  });

    console.log('filteredIcons', filteredIcons)

return (

  <Manager>
  <Input autofocus placeholder="Search" onChange={function (e) {
      setSearch(e.target.value);
    }}
  />

    <br />
    <br />


    <IconGallery>
      {filteredIcons.map((icon) => (
        <IconItem key={icon} name={icon.replace("Icon", "")}>
          <Icon>{icons[icon]()}</Icon>
        </IconItem>
      ))}

      {!filteredIcons.length && (
        <p>No icons were found for "<b>{search}</b>".</p>
      )}
    </IconGallery>

  </Manager>

)
}
