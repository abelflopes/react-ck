import { Meta, Title, IconGallery, IconItem } from "@storybook/blocks";
import { Icon } from "@rck/icon/src";
import icons from "@rck/icon/src/icons";
import { Input } from "@rck/input/src";
import { ThemeProvider } from "@rck/theme/src";

<Meta title="Core/Icons" />

export function Demo ()  {
  const [search, setSearch] = React.useState("");
  const filteredIcons = Object.keys(icons).filter(function (i) {
    return i.includes(search)
  });

return (

  <ThemeProvider>
  <Input autofocus placeholder="Search" onChange={function (e) {
      setSearch(e.target.value);
    }}
  />

    <br />
    <br />


    <IconGallery>
      {filteredIcons.map((icon) => (
        <IconItem key={icon} name={icon}>
          <Icon name={icon} />
        </IconItem>
      ))}

      {!filteredIcons.length && (
        <p>No icons were found for "<b>{search}</b>".</p>
      )}
    </IconGallery>

  </ThemeProvider>

)
}
